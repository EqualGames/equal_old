File: main.cpp

#include "config.hpp"
#include <equal/engine.hpp>
#include <equal/entrypoint.hpp>
#include "scene/main_menu.hpp"
void eq::create_game() {
WindowOptions opt;
opt.vsync = false;
opt.fps_limit = 0;
eq::create_application(opt);
eq::LoadScene<MainMenu>();
}
