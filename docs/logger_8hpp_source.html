161  template <typename... Args>
162  void log(Level level, Args... args) {
163  if (level >= m_level) {
164  std::time_t t = std::time(nullptr);
165  std::tm tm = *std::localtime(&t);
166 
167  std::stringstream ss;
168  ss << std::put_time(&tm, "%Y/%m/%d %H:%M:%S %z");
169 
170  std::cout << "[" << ss.str() << "] [" << level_to_str(level) << "] " << out::LIGHT_GRAY(out::BOLD(m_name)) << " "
171  << fmt::format(args...) << std::endl;
172  }
173  }
